    setupInteriorProps() {
        this.props = [
            { x: 100, y: 350, type: 'EXTINGUISHER' },
            { x: 860, y: 350, type: 'EXTINGUISHER' },
            { x: 480, y: 50, type: 'EMERGENCY_LIGHT' },

    runStep() {
        if (this.stepIndex >= this.script.length) return;
        const step = this.script[this.stepIndex];
        this.uiDialog.style.display = 'none';
        this.uiNarrative.style.display = 'none';

        switch(step.type) {
            case 'SCENE':
                this.scene = step.bg;
                this.camera.targetX = step.cam.x;
                this.camera.targetY = step.cam.y;
                if (step.action) step.action();
                this.nextStep(); 
                break;
            case 'CAMERA_MOVE':
                this.camera.targetX = step.x;
                this.camera.targetY = step.y;
                this.nextStep();
                break;
            case 'FADE_OUT':
                this.fade = { opacity: 0, state: 'OUT', speed: 0.02 };
                const checkFadeOut = () => {
                    if(this.fade.opacity < 1) { requestAnimationFrame(checkFadeOut); }
                    else { this.nextStep(); }
                };
                checkFadeOut();
                break;
            case 'FADE_IN':
                this.fade = { opacity: 1, state: 'IN', speed: 0.02 };
                const checkFadeIn = () => {
                    if(this.fade.opacity > 0) { requestAnimationFrame(checkFadeIn); }
                    else { this.nextStep(); }
                };
                checkFadeIn();
                break;
            case 'ACTION':
                if (step.action) step.action();
                this.nextStep();
                break;
            case 'DIALOG':
                this.uiDialog.style.display = 'block';
                this.elSpeaker.innerText = step.speaker || "";
                this.elText.innerText = step.text;
                this.waitingForInput = true;
                break;
            case 'MOVE':
                this.handleMovement(step);
                break;
            case 'WAIT':
                setTimeout(() => this.nextStep(), (step.frames / 60) * 1000);
                break;
            case 'FADE_TO_BLACK':
                this.fade = { opacity: 0, state: 'OUT_FINAL', speed: 0.01 };
                setTimeout(() => this.nextStep(), 3000);
                break;
            case 'END':
                this.state = 'END';
                this.elText.innerText = "FIN DE LA TRANSMISIÓN";
                this.elSpeaker.innerText = "";
                this.uiDialog.style.display = 'block';
                break;
            case 'ANIMATION':
                if (step.anim === 'HUG') { this.alexo.state = 'IDLE'; this.kid.x = this.alexo.x; this.kid.owner = 'ALEXO'; }
                if (step.anim === 'RESCUER_ARMS') { if(this.rescuers[0]) this.rescuers[0].armsUp = true; }
                if (step.anim === 'EXIT') { this.alexo.facing = 1; this.alexo.state = 'WALK'; }
                this.nextStep();
                break;
            case 'TRANSFER_KID':
                this.kid.owner = 'FIGHTER'; 
                this.alexo.state = 'IDLE'; 
                this.nextStep();
                break;
            case 'EFFECT':
                if (step.name === 'SHAKE') this.camera.shake = 20;
                this.nextStep();
                break;
            case 'SPAWN_RESCUERS':
                // Equipo rescatando: 3 personas
                this.rescuers = [
                    { x: 900, y: 300, armsUp: false, surprised: false }, 
                    { x: 930, y: 310, armsUp: false, surprised: false },
                    { x: 870, y: 290, armsUp: false, surprised: false }
                ];
                this.nextStep();
                break;
            case 'MOVE_RESCUERS':
                this.handleRescuerMove(step.x, step.y, step.speed, () => this.nextStep());
                break;
        }
    }

    handleMovement(step) {
        const actor = step.actor === 'ALEXO' ? this.alexo : this.ian;
        const target = step.x; 
        const animateMove = () => {
            const dx = target - actor.x;
            const dy = (step.y !== undefined ? step.y : actor.y) - actor.y;
            const dist = Math.sqrt(dx*dx + dy*dy);
            if (dist < step.speed) {
                actor.x = target; actor.y = step.y !== undefined ? step.y : actor.y; this.nextStep();
            } else {
                actor.x += (dx/dist) * step.speed; actor.y += (dy/dist) * step.speed;
                requestAnimationFrame(animateMove);
            }
        };
        animateMove();
    }

    handleRescuerMove(targetX, targetY, speed, callback) {
        let allArrived = true;
        this.rescuers.forEach(actor => {
            const dx = targetX - actor.x;
            const dy = targetY - actor.y;
            const dist = Math.sqrt(dx*dx + dy*dy);
            if (dist > speed) {
                actor.x += (dx/dist) * speed; actor.y += (dy/dist) * speed;
                allArrived = false;
            } else {
                actor.x = targetX; actor.y = targetY;
            }
        });
        if (!allArrived) requestAnimationFrame(() => this.handleRescuerMove(targetX, targetY, speed, callback));
        else callback();
    }

    handleInput() {
        if (this.state === 'END') { location.reload(); return; }
        if (this.waitingForInput) {
            this.waitingForInput = false;
            this.uiDialog.style.display = 'none';
            this.stepIndex++;
            this.runStep();
        }
    }

    nextStep() {
        this.stepIndex++;
        this.runStep();
    }

    loop() {
        this.time++;
        this.update();
        this.draw();
        if(this.state !== 'STOPPED') requestAnimationFrame(this.loop);
    }

    update() {
        this.camera.x += (this.camera.targetX - this.camera.x) * 0.05;
        this.camera.y += (this.camera.targetY - this.camera.y) * 0.05;
        if (this.camera.shake > 0) this.camera.shake *= 0.9;
        
        if (this.fade) {
            if (this.fade.state === 'OUT') {
                this.fade.opacity += this.fade.speed;
                if (this.fade.opacity >= 1) this.fade.opacity = 1;
            } else if (this.fade.state === 'IN') {
                this.fade.opacity -= this.fade.speed;
                if (this.fade.opacity <= 0) this.fade.opacity = 0;
            } else if (this.fade.state === 'OUT_FINAL') {
                this.fade.opacity += 0.005;
            }
        }

        this.alexo.frame = Math.sin(this.time * 0.15) * 0.5;
        
        this.seagulls.forEach(s => {
            s.x += s.vx;
            s.y += Math.sin(this.time * 0.05) * 0.5;
            if(s.x > 1000) s.x = -50;
        });

        if (this.ambience.fire > 0 && Math.random() > 0.5) {
            this.particles.push({ x: Math.random() * this.width, y: this.height + 20, vx: (Math.random()-0.5)*2, vy: -Math.random()*3-1, life: 1.0, type: 'FIRE' });
        }
        for (let i = this.particles.length - 1; i >= 0; i--) {
            let p = this.particles[i]; p.x += p.vx; p.y += p.vy; p.life -= 0.02;
            if (p.life <= 0) this.particles.splice(i, 1);
        }
    }

    draw() {
        this.ctx.fillStyle = "#050812";
        this.ctx.fillRect(0, 0, this.width, this.height);

        this.ctx.save();
        const shakeX = (Math.random() - 0.5) * this.camera.shake;
        const shakeY = (Math.random() - 0.5) * this.camera.shake;
        this.ctx.translate(this.width/2 - this.camera.x + shakeX, this.height/2 - this.camera.y + shakeY);

        if (this.scene === 'SALON') this.drawSalon();
        else if (this.scene === 'WINDOW') this.drawWindow();
        else if (this.scene === 'STREET') this.drawStreet();
        else if (this.scene === 'HALL_DARK') this.drawHall(true);
        else if (this.scene === 'APARTMENT') this.drawApartment();
        else if (this.scene === 'STAIRS') this.drawStairs();
        else if (this.scene === 'HALL_LIGHT') this.drawHall(false);
        else if (this.scene === 'EXIT_DOOR') this.drawExitDoor();

        if (this.alexo.visible !== false) this.drawAlexoPro(this.alexo.x, this.alexo.y, this.alexo.state, this.alexo.animState);
        if (this.ian.visible) this.drawIanPro(this.ian.x, this.ian.y, this.ian.holdingTray, this.ian.holdingPaper);
        
        if (this.kid.visible) {
            let kx = this.kid.x;
            let ky = this.kid.y;
            if (this.kid.owner === 'ALEXO') {
                kx = this.alexo.x;
                ky = this.alexo.y; 
            } else if (this.kid.owner === 'FIGHTER' && this.rescuers.length > 0) {
                kx = this.rescuers[0].x;
                ky = this.rescuers[0].y;
            }
            this.drawKidPro(kx, ky, this.kid.state);
        }

        if (this.rescuers.length > 0) this.rescuers.forEach(r => this.drawRescuerPro(r.x, r.y, r.armsUp, r.surprised));

        this.ctx.restore();
        this.drawLighting();
        this.drawFade();
    }

    drawFade() {
        if (this.fade && this.fade.opacity > 0) {
            this.ctx.fillStyle = `rgba(0,0,0,${this.fade.opacity})`;
            this.ctx.fillRect(0, 0, this.width, this.height);
        }
    }

    drawSalon() {

    drawLighting() {
        const ctx = this.ctx;
        let darkness = this.ambience.darkness;
        if (this.scene === 'APARTMENT') darkness = 0.4;
        if (this.scene === 'HALL_DARK') darkness = 0.8;
        
        // Gradiente de oscuridad
        const grad = ctx.createRadialGradient(480, 270, 100, 480, 270, 600);
        grad.addColorStop(0, "rgba(0,0,0,0)"); grad.addColorStop(1, `rgba(0,0,0,${darkness})`);
        ctx.fillStyle = grad;
        ctx.fillRect(0, 0, this.width, this.height);

        // Partículas Fuego
        this.particles.forEach(p => {
            ctx.globalAlpha = p.life;
            if (p.type === 'FIRE') {
                ctx.fillStyle = `rgba(255, ${Math.random()*100 + 50}, 0, ${p.life})`;
                ctx.beginPath(); ctx.arc(p.x, p.y, 5 + Math.random()*5, 0, Math.PI*2); ctx.fill();
            }
            ctx.globalAlpha = 1;
        });

        // MODO INFRARROJO
        if (this.ambience.infrared) {
            ctx.save();
            ctx.globalCompositeOperation = 'multiply';
            ctx.fillStyle = "rgba(0, 50, 0, 0.9)";
            ctx.fillRect(0, 0, this.width, this.height);
            
            ctx.globalCompositeOperation = 'screen';
            ctx.fillStyle = "rgba(255, 100, 50, 0.4)"; 
            ctx.fillRect(600, 50, 200, 200); 
            
            const heatGrad = ctx.createRadialGradient(this.alexo.x, this.alexo.y - 50, 10, this.alexo.x, this.alexo.y - 50, 100);
            heatGrad.addColorStop(0, "rgba(255, 255, 200, 0.6)");
            heatGrad.addColorStop(1, "rgba(0,0,0,0)");
            ctx.fillStyle = heatGrad;
            ctx.fillRect(this.alexo.x - 100, this.alexo.y - 150, 200, 200);
            
            ctx.fillStyle = "rgba(50, 255, 50, 0.2)";
            ctx.fillRect(0,0,this.width, this.height);
            
            ctx.globalCompositeOperation = 'source-over';
            ctx.strokeStyle = "rgba(0, 255, 0, 0.2)";
            ctx.lineWidth = 1;
            for(let i=0; i<this.height; i+=3) {
                ctx.beginPath(); ctx.moveTo(0, i); ctx.lineTo(this.width, i); ctx.stroke();
            }
            
            const lensGrad = ctx.createRadialGradient(480, 270, 150, 480, 270, 600);
            lensGrad.addColorStop(0, "rgba(0,0,0,0)");
            lensGrad.addColorStop(1, "rgba(0,10,0,0.9)");
            ctx.fillStyle = lensGrad;
            ctx.fillRect(0,0,this.width, this.height);
            ctx.restore();
        }
        else if (this.ambience.fire > 0) {
            ctx.globalCompositeOperation = 'overlay';
            ctx.fillStyle = `rgba(255, 50, 0, ${this.ambience.fire * 0.3})`;
            ctx.fillRect(0, 0, this.width, this.height);
            ctx.globalCompositeOperation = 'source-over';
        }
    }

        ctx.fillStyle = "#FFD700"; // Traje de bombero
        ctx.fillRect(-15, -55, 30, 65);
        
        // Casco
        ctx.fillStyle = "#FFD700"; ctx.beginPath(); ctx.arc(0, -65, 14, 0, Math.PI*2); ctx.fill();
        ctx.fillStyle = "#333"; ctx.fillRect(-12, -68, 24, 6); // Visera negra
        
        // OJOS (Lógica sorpresa)
        ctx.fillStyle = "#FFF"; 
        if (surprised) {
            // OJOS MUY GRANDES (Sorpresa)
            ctx.beginPath(); ctx.arc(-5, -65, 6, 0, Math.PI*2); ctx.fill(); // Izq grande
            ctx.beginPath(); ctx.arc(5, -65, 6, 0, Math.PI*2); ctx.fill();  // Der grande
            
            // Cejas altas expresivas
            ctx.strokeStyle = "#333"; ctx.lineWidth=1;
            ctx.beginPath(); ctx.moveTo(-12, -75); ctx.lineTo(-1, -70); ctx.stroke();
            ctx.beginPath(); ctx.moveTo(1, -70); ctx.lineTo(12, -75); ctx.stroke();
        } else {
            // Ojos normales
            ctx.beginPath(); ctx.arc(-5, -65, 2, 0, Math.PI*2); ctx.fill();
            ctx.beginPath(); ctx.arc(5, -65, 2, 0, Math.PI*2); ctx.fill();
        }

        // Brazos
        ctx.fillStyle = "#FFD700";
        if (armsUp) {
            // Recibiendo al niño
            ctx.save(); ctx.translate(-15, -50); ctx.rotate(-0.5); ctx.fillRect(-5, 0, 10, 30); ctx.restore();
            ctx.save(); ctx.translate(15, -50); ctx.rotate(0.5); ctx.fillRect(-5, 0, 10, 30); ctx.restore();
        } else {
            // Brazos normales
            ctx.fillRect(-20, -50, 8, 30); ctx.fillRect(12, -50, 8, 30);
        }
        ctx.restore();
    }
}
